Aeropus QTS – Quick Table Grid Scripts

QTS ermöglicht es dir, Tabellen und Grids schnell und einfach in HTML zu erstellen – direkt im Dokument mit <qts> oder per Inline-Skript.

Anleitung

  1. Füge einen <qts>-Block in dein HTML ein.
  2. Definiere die Gridgröße: grid h:ZEILEN;w:SPALTEN
  3. Fülle die Zellen im <qts-content>-Block. Jede Zeile endet mit ;
  4. Optional: Style bestimmte Zellen mit <qts-style> und Koordinaten h:X;w:Y
  5. Füge das Inline-Skript ein (unten auf der Seite)

Beispiel

<qts>
grid h:2;w:3

<qts-content>
[Name];[Alter];[Status];
[Luković];[27];[Online];
</qts-content>

<qts-style>
coordinate h:1;w:* { font-weight:bold; background:#111; color:white; }
coordinate h:2;w:3 { color:lime; }
</qts-style>
</qts>

Ergebnis

grid h:2;w:3 [Name];[Alter];[Status]; [Luković];[27];[Online]; coordinate h:1;w:* { font-weight:bold; background:#111; color:white; } coordinate h:2;w:3 { color:lime; }

Inline-Skript zum Kopieren


(function () {

  function parseQTS(qtsEl) {
    const raw = qtsEl.innerText;

    const gridMatch = raw.match(/grid\s+h:(\d+);w:(\d+)/i);
    if (!gridMatch) return;
    const h = Number(gridMatch[1]);
    const w = Number(gridMatch[2]);

    const contentMatch = raw.match(/([\s\S]*?)<\/qts-content>/i);
    if (!contentMatch) return;
    const content = contentMatch[1]
      .trim()
      .split("\n")
      .map(r => r.split(";").map(c => c.replace(/[\[\]]/g,"").trim()).filter(c => c.length > 0));

    const grid = document.createElement("div");
    grid.style.display = "grid";
    grid.style.gridTemplateColumns = `repeat(${w}, 1fr)`;
    grid.style.gap = "6px";

    content.forEach((row, y) => {
      row.forEach((cell, x) => {
        const d = document.createElement("div");
        d.className = `qts-cell h${y+1} w${x+1}`;
        d.textContent = cell;
        d.style.padding = "6px 10px";
        d.style.borderRadius = "6px";
        d.style.background = "#1e1e1e";
        d.style.color = "#eee";
        d.style.textAlign = "center";
        grid.appendChild(d);
      });
    });

    const styleMatch = raw.match(/([\s\S]*?)<\/qts-style>/i);
    if (styleMatch) {
      const style = document.createElement("style");
      style.innerHTML = styleMatch[1]
        .replace(/coordinate\s+h:(\*|\d+);w:(\*|\d+)/gi, (_, hh, ww) =>
          `.qts-cell${hh==="*"?"":".h"+hh}${ww==="*"?"":".w"+ww}`
        );
      document.head.appendChild(style);
    }

    qtsEl.replaceWith(grid);
  }

  function parseAllQTS() {
    document.querySelectorAll("qts").forEach(parseQTS);
  }

  if (document.readyState === "loading") {
    document.addEventListener("DOMContentLoaded", parseAllQTS);
  } else {
    parseAllQTS();
  }

})();
